hpstr

201 void EventProcessor::parseVTPData(EVENT::LCGenericObject* vtp_data_lcio)
202 {
203  // First Clear out all the old data.
204  //std::cout << "New VTP Block of size " << vtp_data_lcio->getNInt() << std::endl;
205  vtpData->Clear();
206  for(int i=0; i<vtp_data_lcio->getNInt()/2; ++i)
207  {
208  int data = vtp_data_lcio->getIntVal(i);
209  int secondWord = vtp_data_lcio->getIntVal(i+1);
210  if(!(data & 1<<31)) continue;
211  int type = (data>>27)&0x0F;
212  int subtype;
213  switch (type)
214  {
215  case 0: // Block Header
216  vtpData->blockHeader.blocklevel = (data )&0x00FF;
217  vtpData->blockHeader.blocknum = (data >> 8)&0x03FF;
218  vtpData->blockHeader.nothing = (data >> 18)&0x00FF;
219  vtpData->blockHeader.slotid = (data >> 22)&0x001F;
220  vtpData->blockHeader.type = (data >> 27)&0x000F;
221  vtpData->blockHeader.istype = (data >> 31)&0x0001;
222  //std::cout << i << " BlockHeader " << vtpData->blockHeader.type << std::endl;
223  break;
224  case 1: // Block Tail
225  vtpData->blockTail.nwords = (data )&0x03FFFFF;
226  vtpData->blockTail.slotid = (data >> 22)&0x000001F;
227  vtpData->blockTail.type = (data >> 27)&0x000000F;
228  vtpData->blockTail.istype = (data >> 31)&0x0000001;
229  //std::cout << i << " BlockTail " << vtpData->blockTail.type << std::endl;
230  break;
231  case 2: // Event Header
232  vtpData->eventHeader.eventnum = (data )&0x07FFFFFF;
233  vtpData->eventHeader.type = (data >> 27)&0x0000000F;
234  vtpData->eventHeader.istype = (data >> 31)&0x00000001;
235  //std::cout << i << " EventHeader " << vtpData->eventHeader.eventnum << std::endl;
236  break;
237  case 3: // Trigger time
238  vtpData->trigTime = (data & 0x00FFFFFF) + ((secondWord & 0x00FFFFFF )<<24);
239  //std::cout << i << "&" << i+1 << " trigTime = " << vtpData->trigTime << std::endl;
240  i++;
241  break;
242  case 12: // Expansion type
243  subtype = (data>>23)&0x0F;
244  switch(subtype){
245  case 2: // HPS Cluster
246  VTPData::hpsCluster clus;
247  clus.X = (data )&0x0003F;
248  // If the first bit of the index is 1, then it is a negative number
249  if((clus.X >> 5 & 0x1) == 0x1) clus.X = -((clus.X ^ 0x3F) + 1);
250  clus.Y = (data >> 6)&0x0000F;
251  // If the first bit of the index is 1, then it is a negative number
252  if((clus.Y >> 3 & 0x1) == 0x1) clus.Y = -((clus.Y ^ 0xF) + 1);
253  clus.E = (data >> 10)&0x01FFF;
254  clus.subtype = (data >> 23)&0x0000F;
255  clus.type = (data >> 27)&0x0000F;
256  clus.istype = (data >> 31)&0x00001;
257  clus.T = (secondWord )&0x003FF;
258  clus.N = (secondWord >> 10)&0x0000F;
259  clus.nothing = (secondWord >> 14)&0x3FFFF;
260  vtpData->clusters.push_back(clus);
261  //std::cout << i << "&" << i+1 << " HPS Cluster " << clus.E << std::endl;
262  i++;
263  break;
264  case 3: // HPS Single Trigger
265  VTPData::hpsSingleTrig strig;
266  strig.T = (data )&0x003FF;
267  strig.emin = (data >> 10)&0x00001;
268  strig.emax = (data >> 11)&0x00001;
269  strig.nmin = (data >> 12)&0x00001;
270  strig.xmin = (data >> 13)&0x00001;
271  strig.pose = (data >> 14)&0x00001;
272  strig.hodo1c = (data >> 15)&0x00001;
273  strig.hodo2c = (data >> 16)&0x00001;
274  strig.hodogeo = (data >> 17)&0x00001;
275  strig.hodoecal = (data >> 18)&0x00001;
276  strig.topnbot = (data >> 19)&0x00001;
277  strig.inst = (data >> 20)&0x00007;
278  strig.subtype = (data >> 23)&0x0000F;
279  strig.type = (data >> 27)&0x0000F;
280  strig.istype = (data >> 31)&0x00001;
281  //std::cout << i << " HPS Single Trigger " << strig.subtype << std::endl;
282  vtpData->singletrigs.push_back(strig);
283  break;
284  case 4: // HPS Pair Trigger
285  VTPData::hpsPairTrig ptrig;
286  ptrig.T = (data )&0x003FF;
287  ptrig.clusesum = (data >> 10)&0x00001;
288  ptrig.clusedif = (data >> 11)&0x00001;
289  ptrig.eslope = (data >> 12)&0x00001;
290  ptrig.coplane = (data >> 13)&0x00001;
291  ptrig.dummy = (data >> 14)&0x0001F;
292  ptrig.topnbot = (data >> 19)&0x00001;
293  ptrig.inst = (data >> 20)&0x00007;
294  ptrig.subtype = (data >> 23)&0x0000F;
295  ptrig.type = (data >> 27)&0x0000F;
296  ptrig.istype = (data >> 31)&0x00001;
297  //std::cout << i << " HPS Pair Trigger " << ptrig.subtype << std::endl;
298  vtpData->pairtrigs.push_back(ptrig);
299  break;
300  case 5: // HPS Calibration Trigger
301  VTPData::hpsCalibTrig ctrig;
302  ctrig.T = (data )&0x003FF;
303  ctrig.reserved = (data >> 10)&0x001FF;
304  ctrig.cosmicTrig = (data >> 19)&0x00001;
305  ctrig.LEDTrig = (data >> 20)&0x00001;
306  ctrig.hodoTrig = (data >> 21)&0x00001;
307  ctrig.pulserTrig = (data >> 22)&0x00001;
308  ctrig.subtype = (data >> 23)&0x0000F;
309  ctrig.type = (data >> 27)&0x0000F;
310  ctrig.istype = (data >> 31)&0x00001;
311  //std::cout << i << " HPS Cal Trigger " << ctrig.subtype << std::endl;
312  vtpData->calibtrigs.push_back(ctrig);
313  break;
314  case 6: // HPS Cluster Multiplicity Trigger
315  VTPData::hpsClusterMult clmul;
316  clmul.T = (data )&0x003FF;
317  clmul.multtop = (data >> 10)&0x0000F;
318  clmul.multbot = (data >> 14)&0x0000F;
319  clmul.multtot = (data >> 18)&0x0000F;
320  clmul.bitinst = (data >> 22)&0x00001;
321  clmul.subtype = (data >> 23)&0x0000F;
322  clmul.type = (data >> 27)&0x0000F;
323  clmul.istype = (data >> 31)&0x00001;
324  //std::cout << i << " HPS Clus Mult Trigger " << clmul.subtype << std::endl;
325  vtpData->clustermult.push_back(clmul);
326  break;
327  case 7: // HPS FEE Trigger
328  VTPData::hpsFEETrig fee;
329  fee.T = (data )&0x003FF;
330  fee.region = (data >> 10)&0x0007F;
331  fee.reserved = (data >> 17)&0x0003F;
332  fee.subtype = (data >> 23)&0x0000F;
333  fee.type = (data >> 27)&0x0000F;
334  fee.istype = (data >> 31)&0x00001;
335  //std::cout << i << " HPS FEE Trigger " << fee.subtype << std::endl;
336  vtpData->feetrigger.push_back(fee);
337  break;
338  default:
339  std::cout << "At " << i << " invalid HPS type: " << type << " subtype: " << subtype << std::endl;
340  break;
341  }
342 
343  break;
344  case 14:
345  std::cout << i << "VTP data type not valid: " << type << std::endl;
346  break;
347  default:
348  std::cout << i << "I was not expecting a VTP data type of " << type << std::endl;
349  break;
350  }
351  }

357 {
358  // Parse out TS header
359  unsigned int headerWord = ts_data_lcio->getIntVal(1);
360  tsData->header.wordCount = (headerWord )&0xFFFF; // 0-15 Word Count
361  tsData->header.test = (headerWord >> 16)&0x00FF; // 16-23 Test Word
362  tsData->header.type = (headerWord >> 24)&0x00FF; // 24-31 Trigger Type
363  // Parse out trigger time and Event Number
364  tsData->T = static_cast<unsigned long>(ts_data_lcio->getIntVal(3)) + ( (static_cast<unsigned long>(ts_data_lcio->getIntVal(4)&0xFFFF)<<32));
365  tsData->EN = static_cast<unsigned long>(ts_data_lcio->getIntVal(2)) + ( (static_cast<unsigned long>(ts_data_lcio->getIntVal(4)&0xFFFF0000)<<16));
366  // Parse out prescaled word
367  tsData->prescaled.intval = ts_data_lcio->getIntVal(5); // Full word
368  tsData->prescaled.Single_0_Top = (tsData->prescaled.intval )&0x001; // 0 Low energy cluster
369  tsData->prescaled.Single_1_Top = (tsData->prescaled.intval >> 1)&0x001; // 1 e+
370  tsData->prescaled.Single_2_Top = (tsData->prescaled.intval >> 2)&0x001; // 2 e+ : Position dependent energy cut
