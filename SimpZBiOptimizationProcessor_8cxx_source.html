359 
360  //step_size defines n% of signal distribution to cut in a given variable
361  std::cout << "step_size_ " << step_size_ << std::endl;
362  double cutFraction = step_size_;
363  if(max_iteration_ > (int)1.0/cutFraction)
364  max_iteration_ = (int)1.0/cutFraction;
365 
366  std::cout << "max iteration: " << max_iteration_ << std::endl;
367  //Iteratively cut n% of the signal distribution for a given Test Cut variable
368  for(int iteration = 0; iteration < max_iteration_; iteration ++){
369  double cutSignal = (double)iteration*step_size_*100.0;
370  cutSignal = round(cutSignal);
371  if(debug_) std::cout << "## ITERATION " << iteration << " ##" << std::endl;
372 
373  //Reset histograms at the start of each iteration
374  //Histograms will change with each iteration.
375  if(debug_) std::cout << "[SimpZBiOptimization]::Resetting Histograms" << std::endl;
376  signalHistos_->resetHistograms1d();
377  signalHistos_->resetHistograms2d();
378  bkgHistos_->resetHistograms1d();
379  bkgHistos_->resetHistograms2d();
380  testCutHistos_->resetHistograms1d();
381  testCutHistos_->resetHistograms2d();
382 
383  //At the start of each iteration, save the persistent cut values
384  for(cut_iter_ it=persistentCutsPtr_->begin(); it!=persistentCutsPtr_->end(); it++){
385  std::string cutname = it->first;
386  double cutvalue = persistentCutsSelector_->getCut(cutname);
387  if(debug_){
388  std::cout << "Saving persistent cut: " << cutname << " " << cutvalue << std::endl;
389  }
390  int cutid = persistentCutsSelector_->getCutID(cutname);
391  processorHistos_->Fill2DHisto("persistent_cuts_hh",(double)cutSignal,
392  (double)cutid,cutvalue);
393  processorHistos_->set2DHistoYlabel("persistent_cuts_hh",cutid,cutname);
394  }
395 
396  //Fill signal variable distributions
397  for(int e=0; e < signalMTT_->GetEntries(); e++){
398  signalMTT_->GetEntry(e);
399 
400  //Apply current set of persistent cuts to all events
401  if(failPersistentCuts(signalMTT_))
402  continue;
403 
404  //Fill Signal variable distributions
405  fillEventHistograms(signalHistos_, signalMTT_);
406  }
407 
408  //Initialize Signal Integrals
409  if(iteration == 0){
410  //Integrate each signal variable distribution.
411  //When iterating Test Cuts, we reference these intial integrals, cutting n% of the original distribution.
412  std::cout << "[SimpZBiOptimization]::Integrating initial Signal distributions" << std::endl;
413  for(cut_iter_ it=testCutsPtr_->begin(); it!=testCutsPtr_->end(); it++){
414  std::string cutname = it->first;
415  std::string var = testCutsSelector_->getCutVar(cutname);
416  initialIntegrals_[var] = signalHistos_->integrateHistogram1D("signal_"+var+"_h");
417  if(debug_)
418  std::cout << "Initial Integral for " << var << " is " << initialIntegrals_[var] << std::endl;
419  }
420  }
421 
422  //Loop over each Test Cut. Cut n% of signal distribution in Test Cut variable
423  if(debug_) std::cout << "Looping over Signal Test Cuts" << std::endl;
424  for(cut_iter_ it=testCutsPtr_->begin(); it!=testCutsPtr_->end(); it++){
425  std::string cutname = it->first;
